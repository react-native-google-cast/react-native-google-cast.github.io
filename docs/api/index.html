useCastChannel

▸ useCastChannel(namespace: string, onMessage?: undefined | function): CastChannel | null

Hook that provides the current CastChannel (may be null).

example

import { useCastChannel } from 'react-native-google-cast'

function MyComponent() {
  const castChannel = useCastChannel(
    'urn:x-cast:com.example.custom',
    useCallback(message => { console.log('Received message', message) }, [])
  )

  // later, for example after pressing a button
  function onPress() {
    if (castChannel) {
      castChannel.sendMessage('...')
    }
  }
}

useCastSession

▸ useCastSession(): CastSession | null

Hook that provides the current CastSession (may be null).

example

import { useCastSession } from 'react-native-google-cast'

function MyComponent() {
  const castSession = useCastSession()

  if (castSession) {
    castSession.client.loadMedia(...)
  }
}

Returns: CastSession | null
